//Start of Task
EBTNodeResult::Type UBTT_HopToPlayer::ExecuteTask(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory)
{
    FHopToPlayerMemory* EnemyMem = CastInstanceNodeMemory(NodeMemory);

    EnemyMem->NavSystem = FNavigationSystem::GetCurrent(this);

    EnemyMem->EnemyActor = Cast(OwnerComp.GetAIOwner()->GetPawn());
    EnemyMem->EnemyAiController = Cast(OwnerComp.GetAIOwner());
    EnemyMem->PlayerActor = Cast(UGameplayStatics::GetPlayerCharacter(GetWorld(), 0));

    if (!EnemyMem->PlayerActor)
    {
        return EBTNodeResult::Failed;
    }

    EnemyMem->SkelMesh = Cast(EnemyMem->EnemyActor)->GetMesh();

    if (HaveReachedPlayer(EnemyMem))
    {
         return EBTNodeResult::Succeeded;
    }
    
    //check if can path find to player using TestPathSync (it is much faster than FindPathSync)
    PathFindToPlayer(EnemyMem);

    return EBTNodeResult::InProgress;
}

//End Task if sudden abort
EBTNodeResult::Type UBTT_HopToPlayer::AbortTask(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory)
{
    FHopToPlayerMemory* EnemyMem = CastInstanceNodeMemory(NodeMemory);

    
    EnemyMem->NavSystem->AbortAsyncFindPathRequest(EnemyMem->NavAsyncReqId);

    EnemyMem->SkelMesh->GetAnimInstance()->Montage_Stop(0.25f, SwarmerData->HoppingMontage);

    //EnemyMem->SkelMesh->GetAnimInstance()->Montage_Play(SwarmerData->StunMontage, 1.0f, EMontagePlayReturnType::MontageLength);

    EnemyMem->EnemyAiController->StopMovement();
    EnemyMem->EnemyAiController->GetBlackboardComponent()->SetValueAsBool("StartMove", false);

    return EBTNodeResult::Aborted;
}

    if (!EnemyMem->PlayerActor || !EnemyMem->bIsPathFindingSuccess || !EnemyMem->SkelMesh->GetAnimInstance())
    {
        FinishLatentTask(OwnerComp, EBTNodeResult::Failed);
        return;
    }

    if (EnemyMem->bIsPaused)
    {
        if (!EnemyMem->PlayerActor)
        {
            EnemyMem->SkelMesh->GetAnimInstance()->Montage_Stop(0.25f, SwarmerData->HoppingMontage);
            FinishLatentTask(OwnerComp, EBTNodeResult::Failed);
            return;

            return;

    }

        if (EnemyMem->SkelMesh->GetAnimInstance()->Montage_GetCurrentSection(SwarmerData->HoppingMontage) == "StartMove" && !EnemyMem->bDoOnce)

            EnemyMem->EnemyAiController->RequestMove(EnemyMem->MoveReq, EnemyMem->Path);
            EnemyMem->bDoOnce = true;
        }
        else if (EnemyMem->SkelMesh->GetAnimInstance()->Montage_GetCurrentSection(SwarmerData->HoppingMontage) == "EndMove")
        {
            EnemyMem->bIsPaused = true;
            EnemyMem->EnemyAiController->StopMovement();

        //    //if (EnemyMem->MoveTimeElapsed >= SwarmerData->MoveDuration)

void UBTT_HopToPlayer::OnTaskFinished(UBehaviorTreeComponent& OwnerComp, uint8* NodeMemory, EBTNodeResult::Type TaskResult)
{
    Super::OnTaskFinished(OwnerComp, NodeMemory, TaskResult);

    FHopToPlayerMemory* EnemyMem = CastInstanceNodeMemory(NodeMemory);

    if (EnemyMem)
    {
        FMemory::Memzero(EnemyMem, sizeof(FHopToPlayerMemory));
    }
}

{
    InitializeNodeMemory(NodeMemory, InitType);
}

}

}


//A delegate where it gets called after findpathasync is done. This gets call automatically. It checks whether pathfinding was success or not
void UBTT_HopToPlayer::OnPathFound(uint32 RequestID, ENavigationQueryResult::Type Result, TSharedPtr Path, FHopToPlayerMemory* TaskMemory)
{
    TaskMemory->bIsPathFinding = false;

    if (!TaskMemory->EnemyActor || !TaskMemory->PlayerActor || !TaskMemory->SkelMesh->GetAnimInstance())
    {
        return;
    }

    //if path result is success
    if (Result == ENavigationQueryResult::Success && Path.IsValid() && !Path->IsPartial())
    {
        //UE_LOG(LogTemp, Log, TEXT("Async path found with %d points"), Path->GetPathPoints().Num());
        SetNavigationRequest(TaskMemory);

        if (TaskMemory->EnemyAiController)
        {
            TaskMemory->bIsPathFindingSuccess = true;
            TaskMemory->Path = Path;
            //TaskMemory->EnemyAiController->RequestMove(TaskMemory->MoveReq, Path);
            TaskMemory->SkelMesh->GetAnimInstance()->Montage_Play(SwarmerData->HoppingMontage, 1.0f, EMontagePlayReturnType::Duration);
            TaskMemory->SkelMesh->GetAnimInstance()->Montage_Pause(SwarmerData->HoppingMontage);
        }
    }
    else // if path cannot be tranversed or partial
    {
        TaskMemory->bIsPathFindingSuccess = false;

        TaskMemory->EnemyAiController->StopMovement();

        if (TaskMemory->SkelMesh->GetAnimInstance()->GetActiveInstanceForMontage(SwarmerData->HoppingMontage))
        {
            TaskMemory->SkelMesh->GetAnimInstance()->Montage_Stop(0.25f, SwarmerData->HoppingMontage);
        }

        //UE_LOG(LogTemp, Warning, TEXT("Async pathfinding failed"));
    }
}


//Return a true if reached player, else returns false if not yet reached
bool UBTT_HopToPlayer::HaveReachedPlayer(FHopToPlayerMemory* TaskMemory)
{
    const float DistanceToGoal = FVector::Dist(TaskMemory->EnemyActor->GetActorLocation(), 
        EnableWander ? TaskMemory->EnemyAiController->GetBlackBoardComp()->GetValueAsVector(WanderGoalKey.SelectedKeyName) : TaskMemory->PlayerActor->GetActorLocation());

    TaskMemory->GoalCapsuleRadius = ReachTestIncludeGoalRadius ? TaskMemory->PlayerActor->FindComponentByClass()->GetScaledCapsuleRadius() : 0.0f;
    TaskMemory->AiCapsuleRadius = ReachTestIncludeAgentRadius ? TaskMemory->EnemyActor->FindComponentByClass()->GetScaledCapsuleRadius() : 0.0f;

    //if player reached goal location return true else return false
    if (DistanceToGoal <= AcceptanceRadius + TaskMemory->GoalCapsuleRadius + TaskMemory->AiCapsuleRadius)
    {
        TaskMemory->SkelMesh->GetAnimInstance()->Montage_Stop(0.25f, SwarmerData->HoppingMontage);
        return true;
    }

    return false;
}

//Initialize all the neccsary things for pathfinding and call FindPathAsync()
void UBTT_HopToPlayer::PathFindToPlayer(FHopToPlayerMemory* TaskMemory)
{
    FPathFindingQuery Query(TaskMemory->EnemyActor, *TaskMemory->NavSystem->GetDefaultNavDataInstance(), TaskMemory->EnemyActor->GetActorLocation(), EnableWander ? 
        TaskMemory->EnemyAiController->GetBlackBoardComp()->GetValueAsVector(WanderGoalKey.SelectedKeyName) : TaskMemory->PlayerActor->GetActorLocation());

    Query.bAllowPartialPaths = AllowPartialPath;

    TaskMemory->PathFindingQuery = Query;

    TaskMemory->NavProp = TaskMemory->EnemyAiController->GetNavAgentPropertiesRef();
    FPathFindingResult PathResult;

    //MyMemory->path = UNavigationSystemV1::FindPathToActorSynchronously(GetWorld(), MyMemory->Enemy->GetActorLocation(), MyMemory->Player);
    TaskMemory->FoundPathDelegate.BindUObject(this, &UBTT_HopToPlayer::OnPathFound, TaskMemory);
    TaskMemory->NavAsyncReqId = TaskMemory->NavSystem->FindPathAsync(TaskMemory->NavProp, TaskMemory->PathFindingQuery, TaskMemory->FoundPathDelegate, EPathFindingMode::Hierarchical);
    TaskMemory->bIsPathFinding = true;
}

//Intialize all the neccesart things for move request
void UBTT_HopToPlayer::SetNavigationRequest(FHopToPlayerMemory* TaskMemory)
{
    if (EnableWander)
    {
        TaskMemory->MoveReq.SetGoalLocation(TaskMemory->EnemyAiController->GetBlackBoardComp()->GetValueAsVector(WanderGoalKey.SelectedKeyName));
    }
    else
    {
        TaskMemory->MoveReq.SetGoalActor(TaskMemory->PlayerActor);
        TaskMemory->MoveReq.SetGoalLocation(TaskMemory->PlayerActor->GetActorLocation());
    }

    TaskMemory->MoveReq.SetAllowPartialPath(AllowPartialPath);
    TaskMemory->MoveReq.SetRequireNavigableEndLocation(RequreNaviEndLocation);
    TaskMemory->MoveReq.SetCanStrafe(CanStrafe);
    TaskMemory->MoveReq.SetAcceptanceRadius(AcceptanceRadius);
    TaskMemory->MoveReq.SetUsePathfinding(true);
}
