#include "HealthSystem/HealthSystem.h"
#include "GameFramework/Character.h"
#include "Kismet/GameplayStatics.h"
#include "MindslayerInstance.h"

// Sets default values for this component's properties
UHealthSystem::UHealthSystem()
{
	// Set this component to be initialized when the game starts, and to be ticked every frame.  You can turn these features
	// off to improve performance if you don't need them.
	PrimaryComponentTick.bCanEverTick = false;

	Health = DefaultHealth;
}

void UHealthSystem::CheckHealth(TObjectPtr DamageCauser, TypeDamage Type)
{
	if (Health <= 0)
	{
		UMindslayerInstance* instance = Cast(UGameplayStatics::GetGameInstance(this)); // Passes in score if the enemy is killed
		if (instance)
		{
			float newScore = killScore;

#pragma region Conditionals
			// I'm an amazing programmer

			if (inAir)
			{
				newScore *= inAirMult;
				instance->inAir = true;
			}
			else
			{
				instance->inAir = false;
			}

			if (Type == TypeDamage::PlayerMelee)
			{
				newScore *= isMeleeMult;
				instance->isMelee = true;
			}
			else
			{
				instance->isMelee = false;
			}
#pragma endregion

			instance->AddScore(newScore);
			instance->latestScore = newScore;

			if (BP_Score)
			{
				UWorld* World = GetWorld();
				FActorSpawnParameters SpawnParams;
				SpawnParams.Owner = this->GetOwner();
				SpawnParams.SpawnCollisionHandlingOverride = ESpawnActorCollisionHandlingMethod::AlwaysSpawn;
				World->SpawnActor(BP_Score->GeneratedClass, this->GetOwner()->GetActorLocation(), this->GetOwner()->GetActorRotation(), SpawnParams);
			}
		}
		else
		{
			GEngine->AddOnScreenDebugMessage(-1, 15.0f, FColor::Red, "Mindslayer Instance not found, score systeme will not work!");
		}

		OnZeroHealth.Broadcast(Type);
		//Die();
	}
}

// Called when the game starts
void UHealthSystem::BeginPlay()
{
	Super::BeginPlay();

	AActor* Owner = GetOwner();

	Health = DefaultHealth;
	if (Owner)
	{
		OnTakeDamage.AddDynamic(this, &UHealthSystem::TakeDamage);
		OnHeal.AddDynamic(this, &UHealthSystem::Heal);
	}
	// ...
	
}

void UHealthSystem::TakeDamage(int Damage, AActor* DamageCauser, AController* InstigatedBy, TypeDamage type)
{
	if (Damage <= 0)
	{
		return;
	}

	Health = FMath::Clamp(Health - Damage, 0, DefaultHealth);

	if (type == TypeDamage::OutOfBounds)
	{
		bTakeDamageOutOfBounds = true;
	}

	//if (DamageCauser)
	//{
		//UE_LOG(LogTemp, Warning, TEXT("%s dealt %d dmg. Remaining Health: %d"), *DamageCauser->GetActorNameOrLabel(), Damage, Health);

		CheckHealth(Cast(DamageCauser),type);
	//}
}

void UHealthSystem::Heal(UHealthSystem* Sys, int HealAmt)
{
	Sys->Health += HealAmt;

	Sys->Health = FMath::Clamp(Sys->Health, 0, Sys->DefaultHealth);

	//UE_LOG(LogTemp, Warning, TEXT("heal %d. Remaining Health: %d"),sys->HealAmount, sys->Health);

}

void UHealthSystem::ResetPlayerHealth(UHealthSystem* Sys)
{
	Heal(Sys, 100.0f);
}

void UHealthSystem::SetTutorialHealth(int HealthValue)
{
	Health = HealthValue;
}
