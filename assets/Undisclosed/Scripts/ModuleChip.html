using Game;
using Sirenix.OdinInspector;
using System.Collections.Generic;
using UnityEngine;

namespace Gameplay
{
    public interface IModuleChip : IRewardDropStack
    {
        Sprite IconSprite { get; }
        string Name { get; }
        string Description { get; }

        float BaseWeight { get; }
        float BonusWeight { get; set; }
        float TotalWeight { get; }
        float ChipEffectiveness { get; set; }

        /// 
        /// CurrentLevel starts at zero to read list better but for displaying UI it will always be plus one to it 
        /// 
        public int CurrentLevel { get; set; }
        public List  ModuleLevels { get; }

        int PriorityLevel { get; }

        void Apply(ModuleRarity rarity, ISessionRogue session);

        public ModuleLevel GetNextModuleLevel()
        {
            if(HasReachedMaxLevel())
            {
                return ModuleLevels[CurrentLevel];
            }

            return ModuleLevels[CurrentLevel + 1];
        }

        public ModuleLevel GetCurrentModuleLevel()
        {
            return ModuleLevels[CurrentLevel];
        }

        public ModuleLevel GetPreviousModuleLevel()
        {
            if(CurrentLevel - 1 < 0)
            {
                return ModuleLevels[CurrentLevel];
            }

            return ModuleLevels[CurrentLevel - 1];
        }

        public bool HasReachedMaxLevel()
        {
            if (CurrentLevel >= ModuleLevels.Count - 1)
            {
                return true;
            }

            return false;
        }
        [System.Serializable]
        public class ModuleLevel
        {
            /// 
            /// For display only 
            /// 
            [ReadOnly]
            public int Level;

            [Unit(Units.Percent)]
            [MinValue(0)] public float IncreaseChance;
            /// 
            /// StartingTier starts at zero to read list better but for displaying UI it will always be plus one to it 
            /// 
            [MinValue(0)] public int StartingTier;
            [MinValue(0)] public int UpgradeReqModule;
            [MinValue(0)] public int UpgradeReqChip;
        }
    }
}

ModuleChip.cs

using System;
using System.Collections;
using System.Collections.Generic;
using System.Linq;
using Game;
using Newtonsoft.Json.Linq;
using Sirenix.OdinInspector;
using UnityEngine;

namespace Gameplay
{
    public abstract class ModuleChip : SerializedScriptableObject, IModuleChip
    {
        [PreviewField(60, ObjectFieldAlignment.Center)]
        [SerializeField] private Sprite _iconSprite;
        [SerializeField] private string _name;
        [SerializeField] private string _description;
        [SerializeField] private float _baseWeight = 10.0f;
        [PropertySpace(0, 10)]
        [Tooltip("Lower number means higher priority when multiple modules share the same condition.")]
        [SerializeField] private int _priorityLevel = 100;

        [PropertyOrder(100)]
        [TableList(AlwaysExpanded = true, HideToolbar = true)]
        [SerializeField]
        private List _moduleLevels = new List();

        [Title("Upgrade Level Stat")]
        [Button("Add New Entry", ButtonSizes.Medium), GUIColor(0.4f, 0.8f, 0.4f)]
        private void AddLevels()
        {
            int nextLevel = _moduleLevels.Any() ? _moduleLevels.Max(s => s.Level) + 1 : 1;
            var newEntry = new IModuleChip.ModuleLevel
            {
                Level = nextLevel,
                IncreaseChance = 0,
                StartingTier = 0,
                UpgradeReqModule = 0
            };
            _moduleLevels.Add(newEntry);

            //update all levels so that is in order
            for (int i = 0; i < _moduleLevels.Count; i++)
            {
                _moduleLevels[i].Level = i + 1;
            }
        }
        public Sprite IconSprite => _iconSprite;
        public string Name => _name;
        public string Description => _description;
        public float BaseWeight => _baseWeight;
        public float BonusWeight { get; set; }
        public float TotalWeight => BaseWeight + BonusWeight;
        public float ChipEffectiveness { get; set; } = 100.0f;

        //CurrentLevel starts at zero to read list better but for displaying UI it will always be plus one to it 
        public int CurrentLevel { get; set; }

        public List ModuleLevels => _moduleLevels;

        public int PriorityLevel => _priorityLevel;

        public Sprite SpriteIcon => _iconSprite;
        public string RewardName => _name;
        public Color BorderColor => SessionData.CurrentRogue.ModuleDatabase.GetByName(_name).Color;

        public void Apply(ModuleRarity rarity, ISessionRogue session)
        {
            OnApply(rarity, session);

            //store the obtained module along with its rarity
            if (!session.ObtainedModuleChips.ContainsKey(rarity))
            {
                session.ObtainedModuleChips.Add(rarity, new List());
            }

            session.ObtainedModuleChips[rarity].Add(this);

            // Create a lookup cache to find matching module types and their priority levels from session
            var modulePriorityLookup = session.ObtainedModuleChips.Values
                .SelectMany(list => list)
                .GroupBy(m => m.GetType())
                .ToDictionary(g => g.Key, g => g.First().PriorityLevel);

            // Sort modifiers by module priority level
            foreach (var kvp in session.ModifiersByCondition)
            {
                kvp.Value.Sort((a, b) =>
                {
                    var typeA = a.GetType().DeclaringType;
                    var typeB = b.GetType().DeclaringType;

                    if (typeA == null || typeB == null)
                        return 0;

                    var hasPriorityA = modulePriorityLookup.TryGetValue(typeA, out int priorityA);
                    var hasPriorityB = modulePriorityLookup.TryGetValue(typeB, out int priorityB);

                    if (hasPriorityA && hasPriorityB)
                        return priorityA.CompareTo(priorityB);

                    return 0;
                });
            }
        }
        public abstract void OnApply(ModuleRarity rarity, ISessionRogue session);



        [Title("Server Sync")]
        [PropertyOrder(200)]
        [InfoBox("This field is just for upload server. Actual rarity please adjust at ModuleChip Database.")]
        [SerializeField] ModuleRarity _serverRarity = ModuleRarity.COMMON;

        [PropertyOrder(200)]
        [SerializeField]
        [ReadOnly]
        [LabelText("Server ID (Cached)")]
        [InfoBox("This is the module's ID on the server. Used for updates when name changes.")]
        private string _cachedServerId = "";

        [PropertyOrder(200)]
        [ShowInInspector]
        [ReadOnly]
        [LabelText("Server Status")]
        private string _serverStatus = "Unknown";

        [PropertyOrder(200)]
        [Button("Sync to Server", ButtonSizes.Large), GUIColor(0.3f, 1f, 0.3f)]
        private void SyncToServer()
        {
            if (Application.isPlaying)
            {
                ServerAPI.Instance.StartCoroutine(SyncModuleData());
            }
            else
            {
                Debug.LogWarning("[ModuleChip] Cannot sync in Edit mode. Enter Play mode first.");
            }
        }

        [PropertyOrder(200)]
        [Button("Check Server Status", ButtonSizes.Medium), GUIColor(0.9f, 0.9f, 0.3f)]
        private void CheckServerStatus()
        {
            if (Application.isPlaying)
            {
                ServerAPI.Instance.StartCoroutine(CheckModuleOnServer());
            }
            else
            {
                Debug.LogWarning("[ModuleChip] Cannot check in Edit mode. Enter Play mode first.");
            }
        }

        /// Get the module type (stack or non-stack)
        protected abstract string GetModuleType();

        /// POST or PATCH based on server status
        private IEnumerator SyncModuleData()
        {
            Debug.Log($"[ModuleChip] Starting smart sync for: {_name}");
            Debug.Log($"[ModuleChip] Cached Server ID: {(_cachedServerId ?? "None")}");

            // Build module data JSON
            JObject moduleData = new JObject
            {
                ["name"] = _name,
                ["description"] = _description,
                ["module_type"] = GetModuleType(),
                ["max_level"] = _moduleLevels.Any() ? _moduleLevels.Max(m => m.Level) : 1,
                ["rarity"] = _serverRarity.ToString(),
                ["base_stats"] = new JObject()
            };

            Debug.Log($"[ModuleChip] Module data to sync:");
            Debug.Log($"[ModuleChip] {moduleData.ToString(Newtonsoft.Json.Formatting.Indented)}");

            bool syncComplete = false;
            bool syncSuccess = false;

            // If we have a cached server ID, try to use it directly
            if (!string.IsNullOrEmpty(_cachedServerId))
            {
                Debug.Log($"[ModuleChip] Using cached server ID: {_cachedServerId}");

                yield return ModuleAPI.PatchModuleById(
                    _cachedServerId,
                    moduleData,
                    json =>
                    {
                        Debug.Log($"[ModuleChip] SUCCESS (PATCH with cached ID)");
                        Debug.Log($"[ModuleChip] Module updated successfully: {_name}");
                        Debug.Log($"[ModuleChip] Server response: {json.ToString(Newtonsoft.Json.Formatting.Indented)}");

                        _serverStatus = "Synced";
                        syncSuccess = true;
                        syncComplete = true;

#if UNITY_EDITOR
                        UnityEditor.EditorUtility.SetDirty(this);
#endif
                    },
                    err =>
                    {
                        Debug.LogWarning($"[ModuleChip] Cached ID failed, falling back to name lookup");
                        Debug.LogWarning($"[ModuleChip] Error: {err}");

                        // Clear invalid cached ID
                        _cachedServerId = "";
                        syncComplete = true;

#if UNITY_EDITOR
                        UnityEditor.EditorUtility.SetDirty(this);
#endif
                    }
                );

                // Wait for completion
                while (!syncComplete)
                {
                    yield return null;
                }

                // If patch with cached ID succeeded, we're done
                if (syncSuccess)
                {
                    Debug.Log($"[ModuleChip] Sync completed successfully for: {_name}");
                    yield break;
                }

                // Reset for fallback attempt
                syncComplete = false;
            }

            // No cached ID or cached ID failed - use smart sync (check by name)
            yield return ModuleAPI.SyncModule(
                moduleData,
                json =>
                {
                    Debug.Log($"[ModuleChip] SUCCESS");
                    Debug.Log($"[ModuleChip] Module synced successfully: {_name}");
                    Debug.Log($"[ModuleChip] Server response: {json.ToString(Newtonsoft.Json.Formatting.Indented)}");

                    // Cache the server ID
                    if (json != null && json["id"] != null)
                    {
                        _cachedServerId = json["id"].ToString();
                        Debug.Log($"[ModuleChip] Cached server ID: {_cachedServerId}");

#if UNITY_EDITOR
                        UnityEditor.EditorUtility.SetDirty(this);
#endif
                    }

                    _serverStatus = "Synced";
                    syncSuccess = true;
                    syncComplete = true;
                },
                err =>
                {
                    Debug.LogError($"[ModuleChip] FAILED");
                    Debug.LogError($"[ModuleChip] Failed to sync {_name}: {err}");

                    _serverStatus = "Failed";
                    syncComplete = true;
                }
            );

            // Wait for completion
            while (!syncComplete)
            {
                yield return null;
            }

            if (syncSuccess)
            {
                Debug.Log($"[ModuleChip] Sync completed successfully for: {_name}");
            }
            else
            {
                Debug.LogError($"[ModuleChip] Sync failed for: {_name}");
            }
        }

        /// Check if this module exists on the server
        private IEnumerator CheckModuleOnServer()
        {
            Debug.Log($"[ModuleChip] Checking server status for: {_name}");
            Debug.Log($"[ModuleChip] Current cached ID: {(_cachedServerId ?? "None")}");

            bool checkComplete = false;

            yield return ModuleAPI.CheckModuleExists(
                _name,
                (exists, moduleId) =>
                {
                    if (exists)
                    {
                        Debug.Log($"[ModuleChip] Module EXISTS on server");
                        Debug.Log($"[ModuleChip] Name: {_name}");
                        Debug.Log($"[ModuleChip] ID: {moduleId}");

                        // Update cached ID if different
                        if (_cachedServerId != moduleId)
                        {
                            Debug.Log($"[ModuleChip] Updating cached ID from '{_cachedServerId}' to '{moduleId}'");
                            _cachedServerId = moduleId;

#if UNITY_EDITOR
                            UnityEditor.EditorUtility.SetDirty(this);
#endif
                        }

                        _serverStatus = $"Exists (ID: {moduleId.Substring(0, 8)}...)";
                    }
                    else
                    {
                        Debug.Log($"[ModuleChip] Module NOT FOUND on server");
                        Debug.Log($"[ModuleChip] Name: {_name}");
                        _serverStatus = "Not on server";

                        // Clear cached ID if module doesn't exist
                        if (!string.IsNullOrEmpty(_cachedServerId))
                        {
                            Debug.Log($"[ModuleChip] Clearing invalid cached ID");
                            _cachedServerId = "";

#if UNITY_EDITOR
                            UnityEditor.EditorUtility.SetDirty(this);
#endif
                        }
                    }

                    checkComplete = true;
                }
            );

            // Wait for completion
            while (!checkComplete)
            {
                yield return null;
            }
        }
    }

}

ModuleChipStack.cs

using Sirenix.OdinInspector;
using System.Collections.Generic;
using UnityEngine;

namespace Gameplay
{
    public abstract class ModuleChipStack : ModuleChip, IModuleChipStack where T : IModuleValueType
    {
        [TableList(AlwaysExpanded = true, ShowIndexLabels = true)]
        [Title("Teir Stack Level")]
        [SerializeField] private List _tiers;
        [SerializeField] private bool _canStackInfinite;

        private int startingTier;
        private int currentTier;

        public void InitializeCurrentTier(int startingValue)
        {
            startingTier = startingValue;
            currentTier = startingTier;
        }

        public void IncreaseCurrentTier()
        {
            if(!_canStackInfinite)
            {
                currentTier++;
            }
        }

        public bool HasReachedMaxTier()
        {
            if (currentTier >= _tiers.Count - 1)
            {
                return true;
            }

            return false;
        }

        public ModuleTiers GetCurrentModuleTier()
        {
            return _tiers[currentTier];
        }

        [System.Serializable]
        public class ModuleTiers
        {
            public T Value;
        }

        protected override string GetModuleType()
        {
            return "stack";
        }
    }

}
