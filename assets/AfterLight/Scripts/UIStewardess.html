using UnityEngine;

public class UIStewardessAnimManager : MonoBehaviour
{
    [SerializeField] private float _minSleepTime = 5.0f;
    [SerializeField] private float _maxSleepTime = 10.0f;

    private AudioSource _audioSource;
    private Animator _uiStewardessAniamtor;
    private float _timer;
    private bool _pauseTimer = false;

    private float _sleepTime;
    // Start is called before the first frame update
    void Start()
    {
        _audioSource = GetComponent();
        _uiStewardessAniamtor = GetComponent();
        RandomizeSleepTime();
    }

    // Update is called once per frame
    void Update()
    {
        if(_uiStewardessAniamtor.GetCurrentAnimatorStateInfo(0).normalizedTime > 1.0f)
        {
            if (_uiStewardessAniamtor.GetCurrentAnimatorStateInfo(0).IsName(GameConstants.STATE_WARNING_STEWARDESS_UI))
            {
                SwitchToReturnAnimation();
            }
            else if (_uiStewardessAniamtor.GetCurrentAnimatorStateInfo(0).IsName(GameConstants.STATE_RETURN_STEWARDESS_UI))
            {
                SwitchToIdleAnimation();

            }
        }

        if (_pauseTimer)
        {
            return;
        }

        _timer += Time.deltaTime;

        if (_timer >= _sleepTime)
        {
            _uiStewardessAniamtor.SetTrigger(GameConstants.TRIGGER_SLEEP);
            _pauseTimer = true;

            _timer = 0.0f;
        }
    }

    public void SwitchToWarningAnimation()
    {
        _pauseTimer = true;
        _timer = 0.0f;
        _uiStewardessAniamtor.Play(GameConstants.STATE_WARNING_STEWARDESS_UI, 0, 0.0f);
    }

    private void SwitchToReturnAnimation()
    {
        _pauseTimer = true;
        _timer = 0.0f;
        _uiStewardessAniamtor.Play(GameConstants.STATE_RETURN_STEWARDESS_UI, 0, 0.0f);
    }

    private void SwitchToIdleAnimation()
    {
        _uiStewardessAniamtor.Play(GameConstants.STATE_IDLE_STEWARDESS_UI, 0, 0.0f);
        _pauseTimer = false;
        RandomizeSleepTime();
    }

    private void RandomizeSleepTime()
    {
        _sleepTime = Random.Range(_minSleepTime, _maxSleepTime);
    }

    public void PlaySfx()
    {
        _audioSource.Play();
    }
}
