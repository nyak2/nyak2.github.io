using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Bowl : Utensil
{
    [Header("BowlSpawns")]
    [SerializeField] private Transform _ovenSpawnTrasnform;
    [SerializeField] private Transform _tableSpawnTrasnform;
    [SerializeField] private WeightListUI _ingredientWeightList;
    [SerializeField] private Sprite _fillNoChip;
    [SerializeField] private Sprite _fillWithChip;

    [SerializeField] private GameManager _gameManager;
    private bool _onScale;
    private RigidbodyConstraints _constraints;
    private bool _isSubscribedToBowl = false;
    private Sprite _emptyBowlSprite;

    private void Awake()
    {
        _constraints = rb.constraints;
        _emptyBowlSprite = SpriteRenderer.sprite;
    }

    public override void OnUpdate()
    {
        if (_gameManager.CurrentScreen == ScreenTypes.CounterTop)
        {
            if(transform.position.x >= 7.0f)
            {
                rb.velocity = Vector3.zero;
                transform.position = _ovenSpawnTrasnform.position;
            }
            return;
        }

        if(_gameManager.CurrentScreen == ScreenTypes.Oven)
        {
            if (transform.position.x <= 14.0f)
            {
                rb.velocity = Vector3.zero;
                transform.position = _tableSpawnTrasnform.position;
            }
            return;
        }

    }
    private void OnCollisionStay(Collision collision)
    {
        if(!_onScale)
        {
            if (collision.collider.gameObject.CompareTag("Tool"))
            {
                if (collision.collider.gameObject.GetComponent() is WeighingScale scale)
                {
                    scale.rb.velocity = Vector3.zero;
                    rb.constraints = RigidbodyConstraints.FreezePositionY;
                    DisableCollider();
                    _ingredientWeightList.EnableUI();
                    _onScale = true;
                    _ingredientWeightList.SubscribeWeightScreen();
                }

                if(collision.collider.gameObject.GetComponent() is BakingTray tray)
                {
                    if(!tray.AlrPlaced)
                    {
                        tray.PlaceDough();
                        tray.rb.velocity = Vector3.zero;
                        SpriteRenderer.sprite = _emptyBowlSprite;
                    }
                }
            }

        }
    }
    void OnTriggerEnter(Collider other)
    {
        if (!_isSubscribedToBowl)
        {
            Ingredient ingredient = other.GetComponent();
            if (ingredient != null)
            {
                // Subscribe to pour events
                ingredient.OnIngredientPoured += HandleIngredientPoured;
                _isSubscribedToBowl = true;
            }
        }
    }

    private void OnTriggerExit(Collider other)
    {
        //if (other.gameObject.CompareTag("Tool"))
        //{
        _ingredientWeightList.DisableUI();
        EnableCollider();
        _onScale = false;
        _ingredientWeightList.UnSubscribeWeightScreen();
        rb.constraints = _constraints;

        Ingredient ingredient = other.GetComponent();
        if (ingredient != null && _isSubscribedToBowl)
        {
            // Unsubscribe from pour events
            ingredient.OnIngredientPoured -= HandleIngredientPoured;
            _isSubscribedToBowl = false;
        }

        //}
    }

    private void HandleIngredientPoured(IngredientEnum ingredientType, float amount)
    {
        if (!_gameManager.CurrentSession.CurrentBowlMixes.TryGetValue(ingredientType, out RecipeMix value))
        {
            RecipeMix newMix = new RecipeMix
            {
                HasMixed = false,
                Order = 0,
                Weight = 0
            };
            newMix.Weight += amount;

            _gameManager.CurrentSession.CurrentBowlMixes.Add(ingredientType, newMix);
            _ingredientWeightList.UpdateIngredientUI(ingredientType, newMix.Weight);
        }
        else
        {
            RecipeMix currentMix = _gameManager.CurrentSession.CurrentBowlMixes[ingredientType];

            currentMix.Weight += amount;

            _gameManager.CurrentSession.CurrentBowlMixes[ingredientType] = currentMix;
            _ingredientWeightList.UpdateIngredientUI(ingredientType, currentMix.Weight);
        }

        if(_gameManager.CurrentSession.CurrentBowlMixes.TryGetValue(IngredientEnum.CHOCOLATE_CHIPS, out RecipeMix valie))
        {
            SpriteRenderer.sprite = _fillWithChip;
        }
        else
        {
            SpriteRenderer.sprite = _fillNoChip;

        }
    }
}

using Sirenix.OdinInspector;
using UnityEngine;

public abstract class Utensil : SerializedMonoBehaviour
{
    [InfoBox("Ingore the Y vector and put the Z instead")]
    [SerializeField] private Vector3 _minLimit;
    [SerializeField] private Vector3 _maxLimit;

    [SerializeField] private Rigidbody _rb;
    [SerializeField] private BoxCollider _boxCollider;

    [SerializeField] private SpriteRenderer _spriteRenderer;
    private int _oriSortOrder;

    public SpriteRenderer SpriteRenderer => _spriteRenderer;
    public Rigidbody rb => _rb;
    private Camera _camera;

    [ReadOnly] public bool _isMoving;

    // Start is called before the first frame update
    void Start()
    {
        _camera = Camera.main;
        _oriSortOrder = _spriteRenderer.sortingOrder;
    }
    void Update()
    {
        ClampPosition();
        OnUpdate();
    }

    public virtual void OnUpdate()
    {

    }

    public virtual void DragMove(Vector2 pos, Transform obj)
    {
        // Create a ray from the camera through the screen position
        Ray ray = _camera.ScreenPointToRay(new Vector3(pos.x, pos.y, 0));

        // Create a plane at the desired height (adjust the y value as needed)
        Plane plane = new Plane(Vector3.up, new Vector3(0, obj.localPosition.y, 0));

        // Find where the ray intersects the plane
        float distance;
        if (plane.Raycast(ray, out distance))
        {
            Vector3 worldPos = ray.GetPoint(distance);

            // Clamp each axis individually
            worldPos.x = Mathf.Clamp(worldPos.x, _minLimit.x, _maxLimit.x);
            worldPos.z = Mathf.Clamp(worldPos.z, _minLimit.z, _maxLimit.z);

            _rb.MovePosition(worldPos);
        }
    }

    public virtual void ClampPosition()
    {
        Vector3 clampedPosition = transform.position;

        // Clamp each axis individually
        clampedPosition.x = Mathf.Clamp(clampedPosition.x, _minLimit.x, _maxLimit.x);
        clampedPosition.z = Mathf.Clamp(clampedPosition.z, _minLimit.z, _maxLimit.z);

        transform.position = clampedPosition;
    }

    public virtual void OnLeftClick()
    {
        _rb.useGravity = false;
        _rb.isKinematic = true;
        _isMoving = true;
    }

    public virtual void OnLeftRelease()
    {
        _rb.useGravity = true;
        _rb.isKinematic = false;
        _isMoving = false;
    }

    public void DisableCollider()
    {
        _boxCollider.isTrigger = true;
        _spriteRenderer.sortingOrder = 1;
    }

    public void EnableCollider()
    {
        _boxCollider.isTrigger = false;
        _spriteRenderer.sortingOrder = _oriSortOrder;
    }

}
