using UnityEngine;

public class ItemShadow : MonoBehaviour
{
    [SerializeField] private Transform _spriteTransform;

    [SerializeField] private float _minScale = 0.25f; // Minimum scale when sprite is high
    [SerializeField] private float _maxScale = 1f;   // Maximum scale when sprite is low
    [SerializeField] private float _maxHeight = 5f;  // Height at which scale becomes minimum

    private Vector3 _originalScale;

    // Start is called before the first frame update
    void Start()
    {
        _originalScale = transform.localScale;
    }
    void Update()
    {
        // Get the sprite's Y position (absolute value to handle negative heights)
        float spriteHeight = Mathf.Abs(_spriteTransform.position.y);

        // Calculate scale factor based on height (0 to 1 range)
        float heightFactor = Mathf.Clamp01(spriteHeight / _maxHeight);

        // Inverse the factor so higher height = smaller scale
        float scaleFactor = 1f - heightFactor;

        // Apply the scale within min/max range
        float targetScale = Mathf.Lerp(_minScale, _maxScale, scaleFactor);

        // Apply the scaled size
        transform.localScale = _originalScale * targetScale;
    }
}
